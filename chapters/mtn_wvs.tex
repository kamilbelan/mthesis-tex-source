\subsection{Euler equations}
\label{sec:EULER_EQUATIONS}

\begin{equations}[single,*]
\cstress = -p \identityM + 2 \mu \symvgrad, 
\end{equations}

where $p = p\left(\rho, \theta\right)$ is the thermodynamic pressure, $\identityM$ is the identity matrix, $\mu$ is the dynamic viscosity and $\symvgrad = \sym \grad \vb{v}$ is the symmetric part of the velocity gradient.

\begin{equations}[single,*]
\partial_t\left(\rho \vb{v}\right) + \divergence{\left(\rho \vb{v} \otimes \vb{v}\right)}= \divergence{\cstress} + \rho \vb{b},
\end{equations}

with $\vb{b}$ being the specific density of the body forces, \textit{e.g.} gravity. 

\begin{equations}[single,*]
\partial_t \rho + \divergence{\left(\rho \vb{v}\right)} = 0.
\end{equations}
