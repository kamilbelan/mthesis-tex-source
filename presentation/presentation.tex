\section{Problem settings}
\label{sec:SETTINGS}

\begin{frame}{Phenomena}
	The presence of complicated topography can have significant influence on the atmospheric state, leading to complicated \textbf{mesoscale phenomena.} Our main goal is to capture \textbf{internal gravity waves}, which influence
	\vspace{1em}
	\begin{columns}[T,onlytextwidth]
		\column{0.45\textwidth}
		\begin{itemize}
			\setlength{\itemsep}{1.5em} 
			\item clear air turbulence,
			\item transport and mixing dynamics,
			\item downhill windstorms.
		\end{itemize}
		\column{0.55\textwidth}
		\centering
		\includegraphics[width=\textwidth]{figures/clouds.jpg}
		\vspace{0.5em}
		{\tiny \textit{Mountain waves. \href{https://unsplash.com/photos/white-clouds-over-mountains-during-daytime-I8CF-ST8WxE}{Source: Marc Thunis}}\par}
	\end{columns}
\end{frame}

\begin{frame}{Stratification}
	The source of all interesting phenomena of our interest is the \textbf{stratification} of some physical quantities - \emph{density, pressure} and mainly \textbf{potential temperature.}

			\textbf{Potential Temperature} $\theta$
			\vspace{0.5em}
			\begin{itemize}
				\setlength{\itemsep}{1.5em}

\section{Mathematical description}
\label{sec:MATH}
\begin{frame}{Governing equations}

	\begin{itemize}
		\item air is modelled as a dry inviscid ideal gas
		\item inside $\Omega \subset \mathbb{R}^2$ a rectangular domain the evolution is given by the \emph{compressible Euler equations}
	\end{itemize}
	\begin{math-block}{System of equations}
		\begin{equations}[columns,*]
			\rho \dv{\vb{v}}{t} &= -\grad p + \rho \vb{g}, \\
			\dv{\rho}{t} + \rho\left(\divergence{\vb{v}}\right) &= 0,\\
			\dv{\theta}{t} &= 0,\\
			p\left(\rho, T\right) &= \rho R T.
		\end{equations}
	\end{math-block}
\end{frame}

\begin{frame}{Boundary Conditions}
	\begin{columns}[T,onlytextwidth]

		\begin{column}{0.5\textwidth}
			The boundary conditions on $\partial \Omega = \Gamma_{\, \text{west} \,} \cup \Gamma_{\, \text{east} \,} \cup \Gamma_{\, \text{bot} \,} \cup \Gamma_{\, \text{top} \,}$ for the velocity field $\vb{v}$ read as
			\begin{math-block}{Velocity boundary conditions}
				\begin{equations}[columns,*]
					\vb{v} &= v_0 \vb{e}_x, \, \text{on} \, \Gamma_{\, \text{west} \,}, \\
					\vb{v} \vdot \vb{n} &= 0, \, \text{on} \, \Gamma_{\, \text{bot} \,}, \\
					\partial_{x} p &= 0, \, \text{on} \, \Gamma_{\, \text{east} \,}, \\

			\vspace{0.5em}

\begin{frame}{Initial conditions}
	The initial conditions are given by an \textbf{isothermal hydrostatic balance} at $T_0 = \SI{250}{\kelvin}$
	\begin{math-block}{Initial conditions}
		\begin{equations}[columns,*]
			\vb{v}\left(0, \vb{x}\right) &= \vb{0},\\
			\rho\left(0, \vb{x}\right) &= \rho_0 \exp\left(-x_2 \frac{g}{R T_0}\right) \\
			\theta\left(0, \vb{x}\right) &= T_0\left(\frac{p\left(\rho_0, T_0\right)}{P_0}\right)^{\frac{R}{c_p}}
		\end{equations}
	\end{math-block}
	from the state equations it follows for the pressure

	\begin{equations}[single,*]
		p\left(0, \vb{x}\right) = \rho_0 R T_0 \exp\left(-x_2 \frac{g}{R T_0}\right).
	\end{equations}

\end{frame}

\section{Numerical solution}
\label{sec:NUMERICS}
\begin{frame}{Smoothed Particle Hydrodynamics}
	The governing equations are solved numericaly using the \textbf{Smoothed Particle Hydrodynamics (SPH)} method
	\begin{itemize}
		\item the fluid is partitioned into \emph{abstract material points}, so called \textbf{particles}
		\item particle $a$ possess its $\vb{v}_a, \rho_a, \theta_a, p_a$ and interact \textbf{locally}
		\item \emph{PDE's} for the continuum become \emph{ODE's} for the particles
		\item main tool is the \textbf{interpolation kernel} $w_h\left(\vb{x}\right)$
	\end{itemize}

	\begin{equations}[columns,*]
		\Phi\qty(t,\vb{x}) = \Phi\qty(t,\vb{x}) \star \delta\qty(\vb{x}) &=\int_{\Omega}\Phi\qty(t,\vb{x}') \delta\qty(\vb{x}-\vb{x}')\dd{\vb{x}'} \Rightarrow \\
		[\Phi]^C_{w_h}\qty(t,\vb{x}) \coloneq \Phi\qty(t,\vb{x}) \star w_h\qty(\vb{x}) &= \int_{\Omega}\Phi\qty(t,\vb{x}')w_h\qty(\vb{x}-\vb{x}')\dd{\vb{x}'}, \\
		\Phi_a(t) \equiv [\Phi]_{w_h}^D(t,\vb{x}_a) &\coloneq \sum_b \Phi_a w_{ab} V_b,\\
		\Phi(t,\vb{x}) &\approx [\Phi]_{w_h}^C\qty(t,\vb{x}) \approx \Phi_a(t)
	\end{equations}
\end{frame}

\begin{frame}{Different formulations}
	\begin{columns}[T,onlytextwidth]
		\column{0.4\textwidth}
		\centering
		\begin{math-block}{Classical SPH}
			\begin{equations}[columns,*]
				\dv{\rho_a}{t} &= \sum_b m_b w'_{ab} \vb{v}_{ab}\vdot \vb{e}_{ab}, \\
				\dv{\vb{v}_a}{t} &= -\sum_b\qty(\frac{p_a^{2}}{\rho_a}+\frac{p_b^{2}}{\rho_b})w'_{ab}\vb{e}_{ab} + \vb{g},
			\end{equations}
		\end{math-block}
		\begin{itemize}
			\item poor conservation properties, $\theta$ has to be computed extra,
			\item not suitable for advanced techniques \textit{e.g.} \emph{adaptive $h$}.
		\end{itemize}

		\column{0.55\textwidth}
		\centering
		\begin{math-block}{$p-\theta$ formulation}
			\begin{equations}[columns,*]
				P_a &= \left(\sum_{b} \frac{m_b}{\tilde{\theta_0}} \theta_b w_{ab}\left(h_a\right)\right)^{\gamma},\\
				\dv{\vb{v}_a}{t} &= - \sum_{b} m_b \frac{\theta_a \theta_b}{\tilde{\theta}_0^{2}}\left(\frac{f_{a}P_a}{P_a^{2/\gamma}} w'_{ab}\left(h_a\right) + \frac{f_{b}P_b}{P_b^{2/\gamma}} w'_{ab}\left(h_b\right)\right)\vb{e}_{ab} + \vb{g},\\
				\theta_a &= \, \text{const} \,
			\end{equations}
		\end{math-block}
		\begin{itemize}
			\item guaranteed conservation properties, uses the primary variable $\theta$ directly
			\item naturally includes formulations with adaptive $h$.
		\end{itemize}
	\end{columns}
\end{frame}

		\column{0.4\textwidth}
		\centering
		\textbf{Summary}
		\begin{itemize}
			\setlength{\itemsep}{1.5em} 
			\item flows around a topography induce internal gravity waves
			\item IGWs play a significant role in mesoscale atmospheric dynamics
			\item large scale hydrostatic balance is problematic for classical SPH 
			\item well-balanced schemes must to be used for non-stationary flows
